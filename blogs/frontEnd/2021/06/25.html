# 二、爬取jd搜索的数据

我们使用 puppeteer 这个插件来爬取数据（这个插件模拟了浏览器操作）

代码实现

const puppeteer = require('puppeteer')
const fs = require('fs')
const {sync} = require('./es')

/**
 * 抓取数据
 * @param {*} size 抓取多少页
 * @param {*} keyword 搜索的关键词
 * @returns
 */
const findData = async (size, keyword) => {
  const browser = await (puppeteer.launch({
    executablePath: puppeteer.executablePath(),
    headless: false
  }))
  let arr = []
  for (let i = 1; i <= size; i++) {
    const targetUrl = `https://search.jd.com/Search?keyword=${keyword}&page=${i}`
    const page = await browser.newPage()
    console.log('进入第几页----------', i)

    page.setViewport({
      width: 1200,
      height: 800
    })
    await page.goto(targetUrl, {
      timeout: 0,
      waitUntil: 'domcontentloaded'
    })
    // 因为图片使用了懒加载 我们需要模拟浏览器滚动操作 等页面加载完毕 然后再抓取数据
    await page.evaluate(() => {
      return new Promise((resolve, reject) => {
        var totalHeight = 0
        var distance = 400
        var timer = setInterval(() => {
          window.scrollBy(0, distance)
          totalHeight += distance
          if (totalHeight >= document.body.scrollHeight) {
            clearInterval(timer)
            resolve()
          }
        }, 500)
      })
    })
    const baseNode = 'div#J_goodsList ul'
    const urls = await page.evaluate(sel => {
      const box = Array.from($(sel).find('li'))
      return box.map(v => {
        const image = $(v).find('.gl-i-wrap .p-img a img').attr('src')
        const price = $(v).find('.gl-i-wrap .p-price strong i').text()
        const name = $(v).find('.gl-i-wrap .p-name a em').text()
        return {image: `https:${image}`, price, name}
      })
    }, baseNode)
    arr.push(...urls)
  }
  // 这里我先录入本地查看一下
  fs.writeFileSync('data.json', JSON.stringify(arr, undefined, 2))
  // 也可以直接写入es里
  // sync(arr)
  browser.close()
  return {msg: '完成'}
}

执行写入ES

用到了 插件 @elastic/elasticsearch

const {Client} = require('@elastic/elasticsearch')
const client = new Client({
  "node": "http://localhost:9200",
  "maxRetries": 5,
  "requestTimeout": 10000
})
const log = console.log

// 同步ES
function sync(data) {
  const body = []
  data.forEach((item) => {
    const header = {_index: `goods`, _type: '_doc'}
    body.push({index: header})
    body.push(item)
  })
  log(body)
  client.bulk({body})
    .then(() => log(`同步成功`))
    .catch((e) => log(`同步失败`, e))
}

module.exports = {
  sync
}

# 三、后端api实现

使用 koa2实现简单api

const koa = require('koa')
const router = require('koa-router')()
const app = new koa()
const {search} = require('../es')

const index = router.get('/', ctx => {
  ctx.response.body = 'hello world'
}).routes()

router.get('/search', async (ctx) => {
  console.log(ctx.request.query)
  const {q, pageNo, pageSize} = ctx.request.query
  const res = await search(q, pageNo, pageSize)
  const {hits} = res
  let data = []
  hits.forEach(it => {
    it._source.name = it.highlight.name[0]
    data.push(it._source)
  })
  const ret = {
    total: res.total.value,
    data
  }
  // 允许跨域
  ctx.set('Access-Control-Allow-Origin', '*')
  ctx.body = ret
})

app.use(index)
app.use(router.routes())

app.listen(3000)

es搜索方法

async function search(keyword, pageNo, pageSize) {
  const result = await client.search({
    index: 'goods',
    body: {
      from: +(pageNo - 1) * pageSize || 0,
      size: +pageSize || 10,
      query: {
        match: {
          name: keyword
        }
      },
      // 高亮
      highlight: {
        boundary_scanner_locale: 'zh_CN',
        fields: {
          name: {
            pre_tags: [
              '<em style="color: red">'
            ],
            post_tags: [
              '</em>'
            ]
          }
        }
      }
    }
  })
  const {hits} = result.body
  return hits
}

# 四、前端实现

具体html不展示了 看下js实现 这里引用了 vue 方便数据渲染

<script src="https://cdn.jsdelivr.net/npm/vue@2.6.14"></script>
<script>
  new Vue({
    el: '#app',
    data: {
      keyword: '',
      total: 0,
      pageNo: 1,
      pageSize: 10,
      products: []
    },
    mounted() {
      // 监听滚动
      window.addEventListener('scroll', this.handleScroll, true)
    },
    // 销毁监听
    beforeDestroy() {
      window.removeEventListener('scroll', this.handleScroll, true)
    },
    methods: {
      // 获取数据
      async getProducts() {
        const res = await fetch(`http://localhost:3000/search?q=${this.keyword}&pageNo=${this.pageNo}&pageSize=${this.pageSize}`)
        const ret = await res.json()
        this.products = this.products.concat(ret.data)
        this.total = ret.total
      },
      // 滚动操作
      handleScroll() {
        let dom = document.querySelector('#app')
        //文档内容实际高度（包括超出视窗的溢出部分）
        let scrollHeight = Math.max(dom.scrollHeight, dom.scrollHeight)
        //滚动条滚动距离
        let scrollTop = document.documentElement.scrollTop || document.body.scrollTop
        //窗口可视范围高度
        let clientHeight = dom.innerHeight || Math.min(dom.clientHeight, dom.clientHeight)
        if (clientHeight + scrollTop >= scrollHeight) {
          if (this.pageNo * this.pageSize > this.total) return
          this.pageNo++
          this.getProducts()
        }
      },
    }
  })
</script>
