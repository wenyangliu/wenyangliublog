服务商微信和支付宝支付(APP)

wenyang的空间 2021-06-24 微信支付支付宝支付

最近在做服务商模式下的APP微信和支付宝支付

遇到的一些坑记录一下

# 微信支付

用的第三方插件 tenpay

代码实现如下

const tenpay = require('tenpay')
const config = {
  appid: '公众号的APPID',
  mchid: '服务商ID',
  partnerKey: '服务商密钥',
  notify_url: '回调地址',
  spbill_create_ip: 'IP地址'
}
const api = tenpay.init(config)
const appParams = {
  out_trade_no: '交易订单号',
  body: '微信支付',
  total_fee: 1,
  sub_appid: '子商户的APPID',
  sub_mch_id: '子商户号ID'
}
api.getAppParams(appParams)
.then(res => console.log(res))
.catch(err => console.log(err))

# 支付宝支付

用的官方插件 alipay-sdk

const AlipaySdk = require('alipay-sdk').default
const {sign} = require('alipay-sdk/lib/util')
const alipaySdk = new AlipaySdk({
  appId: '第三方应用的APPID',
  gateway: 'https://openapi.alipay.com/gateway.do',
  keyType: 'PKCS1',
  signType: 'RSA2',
  privateKey: fs.readFileSync(path.join(__dirname, `文件路径/private-key.pem`), 'ascii'),
  alipayRootCertPath: path.join(__dirname, `文件路径/alipayRootCert.crt`),
  appCertPath: path.join(__dirname, `文件路径/appCertPublicKey_${APPID}.crt`),
  alipayPublicCertPath: path.join(__dirname, `文件路径/alipayCertPublicKey_RSA2.crt`)
})


const payParams = {
  notifyUrl: '回调地址',
  bizContent: {
    out_trade_no: '交易号',
    subject: encodeURIComponent('支付宝支付'),
    total_amount: '100',
    body: encodeURIComponent('支付宝支付'),
    product_code: 'QUICK_MSECURITY_PAY',
  },
  appAuthToken: '第三方应用的token',
}
const signData = sign('alipay.trade.app.pay', payParams, alipaySdk.config)
const {url, execParams} = alipaySdk.formatUrl('', signData)
const payInfo = url + '&biz_content=' + encodeURIComponent(execParams.biz_content)
return payInfo.substr(1)

上面的写法网上找的 之前做过商户号是可以支付没问题 现在报验签错误

换成下面的

const payParams = {
  notifyUrl: '回调地址',
  bizContent: {
    out_trade_no: '交易号',
    subject: '支付宝支付',
    total_amount: '100',
    body: '支付宝支付',
    product_code: 'QUICK_MSECURITY_PAY',
  },
  appAuthToken: '第三方应用的token',
  appId: '第三方应用的APPID',
}
const signData = sign('alipay.trade.app.pay', payParams, alipaySdk.config)
const payInfo = new URLSearchParams(signData).toString()
return payInfo
