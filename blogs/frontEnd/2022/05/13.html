    Login,
    #[at("/chat")]
    Chat,
    #[not_found]
    #[at("/404")]

pub fn switch(route: &Route) -> Html {
    match route {
        Route::Login => html! {<Login />},
        Route::Chat => html! {<Chat />},
        Route::NotFound => html! {<h1>{"404"}</h1>},
    }
}

# 五、Hello WebSockets!

我们聊天应用程序的核心是 WebSockets。 WebSockets 使我们能够在服务器和客户端之间异步发送和接收消息，而不需要来自客户端的不断轮询。此功能使 Websockets 成为我们聊天应用程序的完美核心。让我们继续创建一个 WebSocket 服务，它将处理在我们的应用程序中使用 WebSocket 的所有方面！

  1. 在 src 文件夹下创建一个名为 services 的新文件夹。
  2. 在 services 中添加一个名为 mod.rs 的新文件，其内容如下：
pub mod websocket;
1
  1. 我们的 Websocket 服务将处理以下内容：
  • 监听来自 WebSocket 服务器的传入消息。
  • 将消息写入 WebSocket 服务器。
  • 使用 MPSC（多生产者单消费者）通道与其他组件通信。

在 services 文件夹中添加一个名为 websocket.rs 的新文件，其内容如下：

use futures::{channel::mpsc::Sender, SinkExt, StreamExt};
use reqwasm::websocket::{futures::WebSocket, Message};

use wasm_bindgen_futures::spawn_local;

pub struct WebsocketService {
    pub tx: Sender<String>,
}

impl WebsocketService {
    pub fn new() -> Self {
        let ws = WebSocket::open("ws://127.0.0.1:8080").unwrap();

        let (mut write, mut read) = ws.split();

        let (in_tx, mut in_rx) = futures::channel::mpsc::channel::<String>(1000);

        spawn_local(async move {
            while let Some(s) = in_rx.next().await {
                log::debug!("got event from channel! {}", s);
                write.send(Message::Text(s)).await.unwrap();
            }
        });

        spawn_local(async move {
            while let Some(msg) = read.next().await {
                match msg {
                    Ok(Message::Text(data)) => {
                        log::debug!("from websocket: {}", data);
                    }
                    Ok(Message::Bytes(b)) => {
                        let decoded = std::str::from_utf8(&b);
                        if let Ok(val) = decoded {
                            log::debug!("from websocket: {}", val);
                        }
                    }
                    Err(e) => {
                        log::error!("ws: {:?}", e)
                    }
                }
            }
            log::debug!("WebSocket Closed");
        });

        Self { tx: in_tx }
    }
}

再来分析一下 websocket.rs 的代码

  • **第6-8行：**创建包含 Sender 类型的单个属性的 WebsocketService 结构。 Sender 允许我们在接收者稍后将接收的通道上按发送顺序异步发送消息。 Sender 是可克隆的，这意味着使用该服务的每个组件都可以克隆它并使用它将消息发送回接收者，顾名思义——只能有一个。
  • 第11-46行：new 函数初始化服务，类似于其他语言中的构造函数。该函数执行以下操作：
    • 连接到 WebSocket 服务器（第 12 行）
    • 创建 MPSC 通道（第 16 行）
    • 在当前线程上生成一个新的未来（异步任务），它将侦听 MPSC 通道的接收端并将接收到的消息写入 WebSocket 服务器。这就是组件与我们的服务通信的方式——通过通道发送消息（第 18-23 行）。
    • 在当前线程上产生另一个新的未来来监听来自 WebSocket 服务器的传入消息并将它们注销（第 25-43 行）
    • 最后，该函数返回一个 WebSocketService (Self) 实例，该实例包含通道发送器 (in_tx)。

# 七、EventBus

我们目前正在使用 MPSC 通道在组件和 WebSocket 服务之间进行通信，但顾名思义——我们不能使用相同的方法来更新组件，因为我们需要相反的方向，即多消费者（组件）单个生产者（服务)，不存在。

我们使用红豆杉代理！代理可用于“在组件之间路由消息，而与它们在组件层次结构中的位置无关”（取自 Yew 的官方文档）。这意味着我们可以使用代理的调度程序将消息从 WebSocket 服务发送到任何侦听组件。让我们创建一个事件总线服务，我们将使用它来发送这些消息。

在 services 下创建一个名为 event_bus.rs 的新文件，并将以下内容粘贴到其中：

use serde::{Deserialize, Serialize};
use std::collections::HashSet;
use yew_agent::{Agent, AgentLink, Context, HandlerId};

#[derive(Serialize, Deserialize, Debug)]
pub enum Request {
    EventBusMsg(String),
}

pub struct EventBus {
    link: AgentLink<EventBus>,
    subscribers: HashSet<HandlerId>,
}

impl Agent for EventBus {
    type Reach = Context<Self>;
    type Message = ();
    type Input = Request;
    type Output = String;

    fn create(link: AgentLink<Self>) -> Self {
        Self {
            link,
            subscribers: HashSet::new(),
        }
    }

    fn update(&mut self, _msg: Self::Message) {}

    fn handle_input(&mut self, msg: Self::Input, _id: HandlerId) {
        match msg {
            Request::EventBusMsg(s) => {
                for sub in self.subscribers.iter() {
                    self.link.respond(*sub, s.clone())
                }
            }
        }
    }

    fn connected(&mut self, id: HandlerId) {
        self.subscribers.insert(id);
    }

    fn disconnected(&mut self, id: HandlerId) {
        self.subscribers.remove(&id);
    }
}

fn update(&mut self, _ctx: &Context<Self>, msg: Self::Message) -> bool {
        match msg {
            Msg::HandleMsg(s) => {
                let msg: WebSocketMessage = serde_json::from_str(&s).unwrap();
                match msg.message_type {
                    MsgTypes::Users => {
                        let users_from_message = msg.data_array.unwrap_or_default();
                        self.users = users_from_message
                            .iter()
                            .map(|u| UserProfile {
                                name: u.into(),
                                avatar: format!(
                                    "https://avatars.dicebear.com/api/adventurer-neutral/{}.svg",
                                    u
                                )
                                .into(),
                            })
                            .collect();
                        return true;
                    }
                    MsgTypes::Message => {
                        let message_data: MessageData =
                            serde_json::from_str(&msg.data.unwrap()).unwrap();
                        self.messages.push(message_data);
                        return true;
                    }
                    _ => {
                        return false;
                    }
                }
            }
            Msg::SubmitMessage => {
                let input = self.chat_input.cast::<HtmlInputElement>();
                if let Some(input) = input {
                    //log::debug!("got input: {:?}", input.value());
                    let message = WebSocketMessage {
                        message_type: MsgTypes::Message,
                        data: Some(input.value()),
                        data_array: None,
                    };
                    if let Err(e) = self
                        .wss
                        .tx
                        .clone()
                        .try_send(serde_json::to_string(&message).unwrap())
                    {
                        log::debug!("error sending to channel: {:?}", e);
                    }
                    input.set_value("");
                };
                false
            }
        }
    }
